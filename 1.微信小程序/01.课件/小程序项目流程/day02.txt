		overflow:hidden；
		text-overflow:ellipsis；
		display:-webkit-box；//开启webkit内核的盒模型
		-webkit-box-orient:vertical；  //设置子元素对齐方式
		-webkit-line-clamp:2  //设置为2行文本溢出隐藏
